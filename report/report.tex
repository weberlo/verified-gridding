  \| (k - 1, k-1) \|_2^2 &\leq 1 \\
  (k-1)^2 + (k-1)^2 & \leq 1 \\
  2(k^2 - 2k + 1) &\leq 1 \\

  k^2 - 2k + 1 &\leq 0
